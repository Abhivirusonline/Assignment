Q.1 : Given this array: [3,62,234,7,23,74,23,76,92], Using arrow function, create an array of the numbers greater than 70.

let myFun = (arr) => {
var c=0;
var ar=[];
for(var i=0; i < arr.length;i++){
if(arr[i]>70)
ar[c++]=arr[i];
}
return ar;
};
var ans=myFun([3,62,234,7,23,74,23,76,92]);
console.log(ans);
